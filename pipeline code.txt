pipeline {
    agent any

    tools {
        jdk 'JDK_HOME'       // Define in Jenkins → Manage Jenkins → Tools
        maven 'MAVEN_HOME'   // Define in Tools
        nodejs 'NODE_HOME'   // Define in Tools
    }

    environment {
        EC2_HOST = '2300030379@100.29.48.69'
        BACKEND_PATH = '/home/2300030379/backend'
        FRONTEND_PATH = '/home/2300030379/frontend'
    }

    stages {
        stage('Clone Repository') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/2300030379/fullstack-s201.git'
            }
        }

        stage('Build React Frontend') {
            steps {
                dir('crud_front') {
                    sh 'npm install'
                    sh 'npm run build'
                }
            }
        }

        stage('Package React as WAR') {
            steps {
                sh '''
                    rm -rf crud_front/war_content
                    mkdir -p crud_front/war_content/META-INF crud_front/war_content/WEB-INF
                    cp -r crud_front/dist/* crud_front/war_content/
                    jar -cvf crud_front/frontapp1.war -C crud_front/war_content .
                '''
            }
        }

        stage('Build Spring Boot Backend') {
            steps {
                dir('crud_back') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Deploy to EC2') {
            steps {
                sh """
                    echo "Creating directories on EC2..."
                    ssh -o StrictHostKeyChecking=no ${EC2_HOST} "mkdir -p ${BACKEND_PATH} ${FRONTEND_PATH}"

                    echo "Deploying backend..."
                    scp -o StrictHostKeyChecking=no crud_back/target/cruddemo.war ${EC2_HOST}:${BACKEND_PATH}/

                    echo "Deploying frontend..."
                    scp -o StrictHostKeyChecking=no crud_front/frontapp1.war ${EC2_HOST}:${FRONTEND_PATH}/

                    echo "Restarting backend process..."
                    ssh -o StrictHostKeyChecking=no ${EC2_HOST} \
                      "pkill -f cruddemo.war || true && nohup java -jar ${BACKEND_PATH}/cruddemo.war > ${BACKEND_PATH}/backend.log 2>&1 &"
                """
            }
        }
    }

    post {
        success {
            echo "✅ Build and deployment completed successfully!"
        }
        failure {
            echo "❌ Build or deployment failed — check the logs for details."
        }
    }
}
